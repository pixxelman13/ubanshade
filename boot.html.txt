<script>
const lines = [
  "Initializing URBANshade Secure Node...",
  "Verifying clearance credentials...",
  "Loading Hadal Division modules...",
  "Decrypting classified directories...",
  "WARNING: All actions are monitored.",

function typeLine() {
  if (i < lines.length) {
    terminal.innerHTML += lines[i] + "<br>";
    i++;
    setTimeout(typeLine, 700);
  } else {
    setTimeout(() => {
      window.location.href = "index.html";
    }, 1000);
  }
}
